\section*{Problem 5}
Let $A$ be a real $n \times n$ matrix and suppose that for every positive integer $m$ there exists a real symmetric matrix $B$ such that $2021 B = A^m + B^2$. Prove that $|\det A| \leq 1$.

\section*{Problem 6}
For a prime $p$, let $\mathrm{GL}_2(\mathbb{Z}/p\mathbb{Z})$ be the group of invertible $2 \times 2$ matrices mod $p$, and let $S_p$ be the symmetric group on $p$ elements. Show that there is no injective group homomorphism $\varphi: \mathrm{GL}_2(\mathbb{Z}/p\mathbb{Z}) \to S_p$.

\section*{Problem 7}
Let $D \subseteq \mathbb{C}$ be an open set containing the closed unit disk. Let $f: D \to \mathbb{C}$ be holomorphic, and $p(z)$ a monic polynomial. Prove
\[ |f(0)| \leq \max_{|z| = 1} |f(z) p(z)|. \]

\section*{Problem 8}
Let $n$ be a positive integer. At most how many distinct unit vectors can be selected in $\mathbb{R}^n$ such that from any three of them, at least two are orthogonal?
