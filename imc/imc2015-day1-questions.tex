\section*{Problem 1}
For any integer $n \ge 2$ and two $n \times n$ matrices with real entries $A$, $B$ that satisfy
\[ A^{-1} + B^{-1} = (A+B)^{-1}, \]
prove that $\det(A) = \det(B)$. Does the same conclusion follow for matrices with complex entries?

\section*{Problem 2}
For a positive integer $n$, let $f(n)$ be the number obtained by writing $n$ in binary and replacing every $0$ with $1$ and vice versa. (E.g.\ $n = 23 = 10111_2$, $f(n) = 1000_2 = 8$.) Prove that
\[ \sum_{k=1}^n f(k) \le \frac{n^2}{4}. \]
When does equality hold?

\section*{Problem 3}
Let $F(0) = 0$, $F(1) = 3/2$, and $F(n) = \tfrac{5}{2} F(n-1) - F(n-2)$ for $n \ge 2$. Determine whether or not $\sum_{n=0}^\infty \frac{1}{F(2^n)}$ is a rational number.

\section*{Problem 4}
Determine whether or not there exist $15$ integers $m_1, \dots, m_{15}$ such that
\[ \sum_{k=1}^{15} m_k \cdot \arctan(k) = \arctan(16). \]

\section*{Problem 5}
Let $n \ge 2$, let $A_1, A_2, \dots, A_{n+1}$ be $n+1$ points in $n$-dimensional Euclidean space not lying on the same hyperplane, and let $B$ be a point strictly inside the convex hull of $A_1, \dots, A_{n+1}$. Prove that $\angle A_i B A_j > 90^\circ$ holds for at least $n$ pairs $(i,j)$ with $1 \le i < j \le n+1$.
