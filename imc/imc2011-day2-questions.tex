\section*{Problem 1}
Let $(a_n)_{n=0}^\infty$ be a sequence with $\tfrac{1}{2} < a_n < 1$ for all $n \ge 0$. Define $(x_n)$ by
\[
x_0 = a_0, \qquad x_{n+1} = \frac{a_{n+1} + x_n}{1 + a_{n+1} x_n} \quad (n \ge 0).
\]
What are the possible values of $\lim_{n \to \infty} x_n$? Can such a sequence diverge?

\section*{Problem 2}
An alien race has three genders: male, female, and emale. A \emph{married triple} consists of three persons, one from each gender, who all like each other. Any person is allowed to belong to at most one married triple. Feelings are always mutual.

The expedition has $n$ males, $n$ females, and $n$ emales; every member likes at least $k$ persons of each of the two other genders.
\begin{enumerate}
  \item[a)] Show that if $n$ is even and $k = n/2$, it might be impossible to create even one married triple.
  \item[b)] Show that if $k \ge 3n/4$, it is always possible to create $n$ disjoint married triples.
\end{enumerate}

\section*{Problem 3}
Determine the value of
\[
\sum_{n=1}^\infty \ln\!\left(1 + \frac{1}{n}\right) \cdot \ln\!\left(1 + \frac{1}{2n}\right) \cdot \ln\!\left(1 + \frac{1}{2n+1}\right).
\]

\section*{Problem 4}
Let $f(x)$ be a polynomial with real coefficients of degree $n$. Suppose $\dfrac{f(k) - f(m)}{k - m}$ is an integer for all integers $0 \le k < m \le n$. Prove that $a - b$ divides $f(a) - f(b)$ for all distinct integers $a$ and $b$.

\section*{Problem 5}
Let $F = A_0 A_1 \ldots A_n$ be a convex polygon in the plane. For $1 \le k \le n - 1$, define the operation $f_k$ that replaces $F$ with
\[
f_k(F) = A_0 \ldots A_{k-1} A'_k A_{k+1} \ldots A_n,
\]
where $A'_k$ is the reflection of $A_k$ across the perpendicular bisector of $A_{k-1} A_{k+1}$. Prove that $(f_1 \circ f_2 \circ \cdots \circ f_{n-1})^n (F) = F$, assuming all intermediate polygons are convex.
