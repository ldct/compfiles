\section*{Problem 1}
Let $f$ be a polynomial of degree 2 with integer coefficients. Suppose $5 \mid f(k)$ for every integer $k$. Prove all coefficients of $f$ are divisible by $5$.

\subsection*{Solution}
Consider $f \in \mathbb{F}_5[x]$. It has 5 roots (all of $\mathbb{F}_5$) but degree $\le 2$, so $f \equiv 0 \pmod 5$.

\emph{Direct:} $f(x) = ax^2 + bx + c$. $5 \mid f(0) = c$, $5 \mid f(1) = a + b + c$, $5 \mid f(-1) = a - b + c$. Hence $5 \mid 2a, 2b, c$, so $5 \mid a, b, c$.

\section*{Problem 2}
For $n \ge 2$, find min/max rank of $n \times n$ matrix with entries $\{1, 2, \ldots, n^2\}$.

\subsection*{Solution}
\textbf{Min = 2, max = $n$.}

\emph{Rank $\ge 2$:} WLOG $a_{11} = 1 < a_{21} < \cdots < a_{n1}$ and $a_{11} < a_{12} < \cdots < a_{1n}$. So $a_{n1}, a_{1n} \ge n$ with strict inequality somewhere. Then $\det\begin{pmatrix}a_{11} & a_{1n} \\ a_{n1} & a_{nn}\end{pmatrix} < n^2 - n \cdot n = 0$ (using $a_{11} a_{nn} < n^2$ and $a_{1n} a_{n1} > n \cdot n$)\footnote{Properly: the $2 \times 2$ minor has rank 2.}, so rank $\ge 2$.

\emph{Rank can be 2:} $T_{ij} = (i-1) n + j$. Each row equals $(1, 2, \ldots, n) + n(i-1)(1, \ldots, 1)$, so rows span a 2-dim space.

\emph{Rank can be $n$:} place odd numbers on the diagonal, even numbers above; determinant is odd.

\section*{Problem 3}
Call $P(x_1, \ldots, x_k)$ \emph{good} if there exist $2 \times 2$ real matrices $A_i$ with $P = \det(\sum x_i A_i)$. Find all $k$ for which every homogeneous degree-2 polynomial in $k$ variables is good.

\subsection*{Solution}
Answer: $k \in \{1, 2\}$.

$k = 1$: $P(x) = \alpha x^2 = \det\bigl(x \begin{pmatrix}1 & 0\\0 & \alpha\end{pmatrix}\bigr)$.

$k = 2$: $\alpha x^2 + \beta y^2 + \gamma x y = \det\bigl(x A_1 + y A_2\bigr)$ with $A_1 = \begin{pmatrix}1 & 0\\0 & \alpha\end{pmatrix}$, $A_2 = \begin{pmatrix}0 & \beta\\-1 & \gamma\end{pmatrix}$.

$k \ge 3$: $P = \sum x_i^2$ is not good. The first columns of $A_1, \ldots, A_k$ (in $\mathbb{R}^2$, $k \ge 3$ vectors) are linearly dependent, so $\sum y_i A_i$ has zero first column for some $(y_i) \ne 0$. Then $\det = 0 \ne P(y) = \sum y_i^2$.

\section*{Problem 4}
Finite group $G$ partitioned as $A \sqcup B \sqcup C$. For sets $U, V, W$, let $N_{UVW} = \#\{(x,y,z) \in U \times V \times W : xyz = e\}$. Prove $N_{ABC} = N_{CBA}$.

\subsection*{Solution}
Basic identities:
\begin{enumerate}
  \item $N_{UVG} = |U| \cdot |V|$ (for each $(x, y)$, $z$ is determined).
  \item $N_{UVW} = N_{VWU} = N_{WUV}$ (cyclic).
  \item $N_{UVW}$ additive in each argument under disjoint union.
\end{enumerate}

Compute:
\begin{align*}
N_{ABC} &= N_{ABG} - N_{ABA} - N_{ABB} = |A||B| - N_{BAA} - N_{BAB} \\
&= N_{BAG} - N_{BAA} - N_{BAB} = N_{BAC} = N_{CBA}.
\end{align*}

\section*{Problem 5}
$f : \mathbb{Z} \to \mathbb{R}$ satisfies $\sum_{i=1}^n f(k + a_i \ell) = 0$ for all $k \in \mathbb{Z}, \ell \ne 0$. Prove $f = 0$.

\subsection*{Solution}
Define
\[
\mathcal{I} = \Bigl\{ \sum b_j X^j \in \mathbb{R}[X] : \sum b_j f(k + j \ell) = 0 \; \forall k, \ell \ne 0 \Bigr\}.
\]
$\mathcal I$ is an ideal. $R(X) = \sum X^{a_i} \in \mathcal I$, so $\mathcal I \ne 0$; generated by some $Q$. If $Q$ constant, $f = 0$.

Else $Q$ has a complex root $c$. For any $m \ge 1$, $Q(X^m) \in \mathcal I$ (replace $\ell$ by $m \ell$), so $Q \mid Q(X^m)$, giving $c^m$ is a root of $Q$ for all $m$. Since $Q$ has finitely many roots, $c^N = c^M$ for some $N > M$, so some $c^j = 1$; in particular one root of $Q$ is $1$, hence $Q(1) = 0$. But $R(1) = n \ne 0$, contradiction.

\section*{Problem 6}
How many nonzero coefficients can $P \in \mathbb{Z}[z]$ have if $|P(z)| \le 2$ on $|z| = 1$?

\subsection*{Solution}
Answer: $0, 1, 2$ (examples: $0$, $1$, $1 + z$).

Suppose $P$ has $\ge 2$ nonzero coefficients. WLOG $a_0 > 0$. Let $Q(z) = a_1 z + \cdots + a_{n-1} z^{n-1}$. Choose $w_k$ with $a_n w_k^n = |a_n|$:
\[
w_k = \begin{cases} e^{2k\pi i/n} & a_n > 0 \\ e^{(2k+1)\pi i/n} & a_n < 0 \end{cases}.
\]
Then $\sum_{k=0}^{n-1} Q(w_k) = 0$ (sum of $n$th roots of unity). So
\[
\frac{1}{n}\sum P(w_k) = a_0 + |a_n|.
\]
Hence
\[
2 \ge \frac{1}{n}\sum |P(w_k)| \ge a_0 + |a_n| \ge 2.
\]
So $a_0 = |a_n| = 1$ and $|P(w_k)| = |2 + Q(w_k)| = 2$ for all $k$, meaning $Q(w_k)$ lies on $|2 + z| = 2$ with sum $0$, forcing $Q(w_k) = 0$. Since $Q$ has degree $< n$ but $n$ roots, $Q = 0$. Thus $P = a_0 + a_n z^n$ has 2 nonzero coefficients.
