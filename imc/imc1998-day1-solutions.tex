\section*{Problem 1}
$V$ is a 10-dimensional real vector space, $U_1\subseteq U_2$ with $\dim U_1=3$, $\dim U_2=6$. $\mathcal E$ is the set of linear maps $T:V\to V$ with $T(U_1)\subseteq U_1$, $T(U_2)\subseteq U_2$. Compute $\dim\mathcal E$.

\subsection*{Solution}
Choose a basis of $V$ adapted to $U_1\subset U_2\subset V$. The matrix of $T\in\mathcal E$ is block upper-triangular with blocks $3\times 3$, $3\times 3$, $4\times 4$ along the diagonal and free above-diagonal blocks. The nonzero entries form three rectangular regions: $3\cdot 3 + 6\cdot 3 + 10\cdot 4 = 9 + 18 + 40 = 67$. Thus $\dim\mathcal E = 67$.

\section*{Problem 2}
Prove: For any permutation $\pi_1\ne \mathrm{id}$ of $\{1,\dots,n\}$, there exists $\pi_2$ such that every permutation is a composition of $\pi_1, \pi_2$'s. Holds for $n=3,5$; fails for $n=4$.

\subsection*{Solution}
$n=3$: if $x=(12)$ take $y=(123)$; if $x=(123)$ take $y=(12)$.

$n=4$: Let $x=(12)(34)$. Suppose $\langle x,y\rangle = S_4$. Let $K = \{\mathrm{id}, (12)(34), (13)(24), (14)(23)\}$, the Klein four-group, normal in $S_4$. Then $S_4/K$ is generated by $\bar y = yK$, hence cyclic. But $S_4/K\cong S_3$ is not cyclic. Contradiction.

$n=5$: case analysis by conjugacy class of $x$:
\begin{itemize}
\item $x=(12)$: take $y=(12345)$.
\item $x=(123)$: take $y=(124)(35)$. Then $y^3 x y^3 = (125)$, $y^4 = (124)$. So $(123),(124),(125)\in\langle x,y\rangle$, generating $A_5$; since $y$ is odd, $\langle x,y\rangle = S_5$.
\item $x=(123)(45)$: take $y=(124)$; reduces similarly.
\item $x=(1234)$: take $y=(12345)$. $(yx)^3 = (24)$, $x^2(24)=(13)$, $y^2=(13524)$. So $\langle x,y\rangle = S_5$.
\item $x=(12)(34)$: take $y=(1354)$. Then $y^2 x = (125)$, $y^3 x=(124)(53)$; reduces to previous case.
\item $x=(12345)$: take $y=(12)$.
\end{itemize}

\section*{Problem 3}
$f(x)=2x(1-x)$, $f_n = f\circ\cdots\circ f$.

(a) Find $\lim_n \int_0^1 f_n$.

(b) Compute $\int_0^1 f_n$.

\subsection*{Solution}
(a) For $x_0\in(0,1)$, $x_1\in(0,1/2]$; by induction $x_n\le f(x_n)\le 1/2$, $(x_n)$ bounded increasing to $l$ with $l=2l(1-l)$, so $l=1/2$. Dominated convergence gives $\int_0^1 f_n\to 1/2$.

(b) By induction: $f_n(x) = \frac 12 - 2^{2^n-1}(x-\frac 12)^{2^n}$. Base $n=1$ direct. Induction:
\[f_{k+1}(x) = f_k(f(x)) = \tfrac 12 - 2^{2^k-1}(f(x)-\tfrac 12)^{2^k} = \tfrac 12 - 2^{2^{k+1}-1}(x-\tfrac 12)^{2^{k+1}}.\]
Integrating:
\[\int_0^1 f_n = \tfrac 12 - \tfrac{1}{2(2^n+1)}.\]

\section*{Problem 4}
$f:\mathbb{R}\to\mathbb{R}$ twice differentiable, $f(0)=2$, $f'(0)=-2$, $f(1)=1$. Show $\exists\,\xi\in(0,1)$ with $f(\xi)f'(\xi)+f''(\xi)=0$.

\subsection*{Solution}
Let $g(x) = \tfrac 12 f^2(x) + f'(x)$. Then $g'(x) = f(x)f'(x)+f''(x)$ and $g(0) = \tfrac 12\cdot 4 + (-2) = 0$. We show $g(\eta)=0$ for some $\eta\in(0,1]$; then Rolle gives $\xi$.

Case 1: $f$ never zero. Let $h(x) = \tfrac x2 - \tfrac{1}{f(x)}$. Then $h(0) = -1/2$, $h(1) = 1/2 - 1 = -1/2$. Rolle: $h'(\eta)=0$ for some $\eta$. Now $h' = \tfrac 12 + \tfrac{f'}{f^2} = \tfrac{g}{f^2}$. So $g(\eta)=0$.

Case 2: $f$ has zeros. Let $z_1, z_2$ be first and last (closed set). $f>0$ on $[0,z_1)$, so $f'(z_1)\le 0$, giving $g(z_1)=f'(z_1)\le 0$; similarly $g(z_2)\ge 0$. By IVT, $g(\eta)=0$ for some $\eta\in[z_1,z_2]$.

\section*{Problem 5}
$P$ a real polynomial of degree $n$ with only real zeros.

(a) $(n-1)(P'(x))^2 \ge nP(x)P''(x)$ for all real $x$.

(b) Characterize equality.

\subsection*{Solution}
If $P(x)\ne 0$, let $x_1,\dots,x_n$ be zeros. Then
\[\frac{P'(x)}{P(x)} = \sum_i \frac{1}{x-x_i},\quad \frac{P''(x)}{P(x)} = 2\sum_{i<j}\frac{1}{(x-x_i)(x-x_j)}.\]
Compute:
\[(n-1)\left(\frac{P'}{P}\right)^2 - n\frac{P''}{P} = \sum_i \frac{n-1}{(x-x_i)^2} - \sum_{i<j}\frac{2}{(x-x_i)(x-x_j)} = \sum_{i<j}\left(\frac{1}{x-x_i} - \frac{1}{x-x_j}\right)^2 \ge 0.\]
If $P(x_i)=0$: equality requires $P'(x_i)=0$, i.e. multiple root.

Equality for all $x$ iff all $x_i$ are equal, i.e., $P(x) = c(x-a)^n$.

\section*{Problem 6}
$f:[0,1]\to\mathbb{R}$ continuous with $xf(y)+yf(x)\le 1$ for all $x,y$.

(a) Show $\int_0^1 f \le \pi/4$.

(b) Give an equality case.

\subsection*{Solution}
Substitute $x=\sin\theta$, $y=\cos\theta$: $\int_0^1 f(x)\,dx = \int_0^{\pi/2} f(\sin\theta)\cos\theta\,d\theta = \int_0^{\pi/2}f(\cos\theta)\sin\theta\,d\theta$. Summing:
\[2\int_0^1 f = \int_0^{\pi/2}(\sin\theta\, f(\cos\theta) + \cos\theta\, f(\sin\theta))\,d\theta \le \int_0^{\pi/2}1\,d\theta = \pi/2.\]

(b) $f(x) = \sqrt{1-x^2}$: $xf(y)+yf(x) = \sin(\theta+\varphi) \le 1$, and $\int_0^1 \sqrt{1-x^2} = \pi/4$.
