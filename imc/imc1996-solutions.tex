\section*{Problem 1 (Day 1)}
Let $a_j = a_0 + jd$ for $j = 0, \ldots, n$, where $a_0, d$ are fixed real numbers. Put
\[
A = \begin{pmatrix}
a_0 & a_1 & a_2 & \cdots & a_n \\
a_1 & a_0 & a_1 & \cdots & a_{n-1} \\
a_2 & a_1 & a_0 & \cdots & a_{n-2} \\
\vdots & & & & \vdots \\
a_n & a_{n-1} & a_{n-2} & \cdots & a_0
\end{pmatrix}.
\]
Calculate $\det(A)$.

\subsection*{Solution}
Add the first column of $A$ to the last column: last column becomes $(a_0+a_n)(1,1,\ldots,1)^T$, so
\[
\det(A) = (a_0+a_n) \det\begin{pmatrix}
a_0 & a_1 & \cdots & 1 \\
a_1 & a_0 & \cdots & 1 \\
\vdots & & & \vdots \\
a_n & a_{n-1} & \cdots & 1
\end{pmatrix}.
\]
Subtract row $k$ from row $k+1$ (from bottom to top); entries in the first $n$ columns become $\pm d$, last column becomes $0$ (except top row). Expanding and reducing to lower triangular via row operations yields
\[
\det(A) = (-1)^n (a_0+a_n)\, 2^{n-1} d^n.
\]

\section*{Problem 2 (Day 1)}
Evaluate
\[
\int_{-\pi}^{\pi} \frac{\sin nx}{(1+2^x)\sin x}\,dx,
\]
where $n \in \mathbb{N}$.

\subsection*{Solution}
Let $I_n$ be the integral. Split into $[-\pi,0]$ and $[0,\pi]$; substitute $x \mapsto -x$ on $[-\pi,0]$:
\[
I_n = \int_0^\pi \frac{\sin nx}{(1+2^x)\sin x}dx + \int_0^\pi \frac{\sin nx}{(1+2^{-x})\sin x}dx
= \int_0^\pi \frac{\sin nx}{\sin x}\,dx.
\]
For $n \geq 2$,
\[
I_n - I_{n-2} = \int_0^\pi \frac{\sin nx - \sin(n-2)x}{\sin x}dx = 2\int_0^\pi \cos(n-1)x\,dx = 0.
\]
With $I_0 = 0$, $I_1 = \pi$:
\[
I_n = \begin{cases} 0 & n \text{ even},\\ \pi & n \text{ odd}. \end{cases}
\]

\section*{Problem 3 (Day 1)}
A linear operator $A$ on $V$ with $\dim V = n < \infty$ is an involution if $A^2 = E$.
(i) Prove every involution has a basis of eigenvectors.
(ii) Find the maximum number of distinct pairwise commuting involutions on $V$.

\subsection*{Solution}
(i) Let $B = \tfrac12(A+E)$. Then $B^2 = \tfrac14(A^2 + 2A + E) = \tfrac14(2A+2E) = B$, so $B$ is a projection, diagonalizable with eigenvalues in $\{0,1\}$. Hence $A = 2B - E$ is diagonalizable with eigenvalues $\pm 1$.

(ii) Commuting diagonalizable operators are simultaneously diagonalizable (induct on $\dim V$: pick $A_1$-eigenspace $V_\lambda$, invariant under each $A_i$ since $A_i A_1 = A_1 A_i$). In a common basis, each involution has diagonal entries $\pm 1$, so at most $2^n$ distinct commuting involutions.

\section*{Problem 4 (Day 1)}
Let $a_1 = 1$, $a_n = \frac{1}{n}\sum_{k=1}^{n-1} a_k a_{n-k}$ for $n \geq 2$. Show
(i) $\limsup |a_n|^{1/n} < 2^{-1/2}$;
(ii) $\limsup |a_n|^{1/n} \geq 2/3$.

\subsection*{Solution}
(i) Claim: $a_n \leq q^n$ for $n \geq 3$, with $q = 0.7 < 2^{-1/2}$. Compute $a_2 = 1/2$, $a_3 = 1/3$, $a_4 = 11/48$; verify for $n = 3, 4$. Assume for $n \leq N-1$, $N \geq 5$:
\[
a_N = \tfrac{2}{N}a_{N-1} + \tfrac{1}{N}a_{N-2} + \tfrac{1}{N}\sum_{k=3}^{N-3} a_k a_{N-k}
\leq \tfrac{2}{N}q^{N-1} + \tfrac{1}{N}q^{N-2} + \tfrac{N-5}{N}q^N \leq q^N,
\]
using $2/q + 1/q^2 \leq 5$.

(ii) Claim $a_n \geq q^n$ for $n \geq 2$, $q = 2/3$. $a_2 = 1/2 > (2/3)^2$. Induct:
\[
a_N = \tfrac{2}{N}a_{N-1} + \tfrac{1}{N}\sum_{k=2}^{N-2} a_k a_{N-k} \geq \tfrac{2}{N}q^{N-1} + \tfrac{N-3}{N}q^N = q^N,
\]
since $2/q = 3$.

\section*{Problem 5 (Day 1)}
(i) Let $a, b \in \mathbb{R}$ with $b \leq 0$, $1 + ax + bx^2 \geq 0$ on $[0,1]$. Prove
\[
\lim_{n\to\infty} n \int_0^1 (1+ax+bx^2)^n\,dx = \begin{cases} -1/a & a < 0, \\ +\infty & a \geq 0. \end{cases}
\]
(ii) Let $f:[0,1]\to[0,\infty)$ have $f'' \leq 0$ on $[0,1]$ (continuous). Suppose $L = \lim n\int_0^1 f^n$ exists with $0 < L < \infty$. Prove $f'$ has constant sign and $\min_{[0,1]}|f'| = L^{-1}$.

\subsection*{Solution}
(i) By linear change of variable, reduce to the equivalent: $b\leq 0$, $A>0$, $1+ax+bx^2 > 0$ on $[0,A]$, $I_n = n\int_0^A (1+ax+bx^2)^n dx$. Show $I_n \to -1/a$ if $a<0$, $+\infty$ if $a\geq 0$.

For $a < 0$: $f(x) = e^{ax} - (1+ax+bx^2)$ satisfies $f(0)=f'(0)=0$, $f''(x) = a^2 e^{ax} - 2b \geq 0$, so $0 < e^{ax} - (1+ax+bx^2) < cx^2$ with $c = a^2/2 - b$. By MVT:
\[
0 < e^{anx} - (1+ax+bx^2)^n < cx^2 n e^{a(n-1)x}.
\]
Then $n\int_0^A e^{anx}dx = (e^{anA}-1)/a \to -1/a$, and $n^2\int_0^A x^2 e^{a(n-1)x}dx \to 0$, giving $I_n \to -1/a$.

For $a\geq 0$, for $n > \max(A^{-2}, -b) - 1$:
\[
n\int_0^A (1+ax+bx^2)^n dx > n\int_0^{1/\sqrt{n+1}} (1+bx^2)^n dx > \frac{n}{\sqrt{n+1}}\Big(1 + \frac{b}{n+1}\Big)^n \to \infty.
\]

(ii) Let $M = \max f$. If $M < 1$: $I_n \leq nM^n \to 0$, contradiction. If $M > 1$: $f > 1$ on interval $I$ with $|I|>0$ gives $I_n \geq n|I| \to \infty$, contradiction. So $M = 1$.

Suppose $f'(x_0)=0$, $x_0 \in (0,1)$. Then $f(x_0) = M = 1$. Near $x_0$, $f(x_0+h) \geq 1 + \tfrac{m}{2}h^2$ with $m = \min f''$. Pick $\delta$ with $1 + \tfrac{\delta^2}{2}m > 0$, $x_0 + \delta < 1$:
\[
I_n \geq n\int_0^\delta \Big(1+\tfrac{m}{2}h^2\Big)^n dh \to \infty
\]
by (i$'$) â€” contradiction. So $f$ is monotone, $M = f(0)$ or $f(1)$.

Case $M = f(0) = 1$: $f'(0) \neq 0$ (else apply the above argument). $f$ decreasing, $f'(0) < 0$. For $h \in [0,1]$:
\[
1 + hf'(0) \geq f(h) \geq 1 + hf'(0) + \tfrac{m}{2}h^2.
\]
Pick $A < 1$ with $1 + Af'(0) + \tfrac{m}{2}A^2 > 0$; both $n\int_0^A (1+hf'(0))^n dh$ and $n\int_0^A (1 + hf'(0) + \tfrac{m}{2}h^2)^n dh$ tend to $-1/f'(0)$ by (i$'$), so $n\int_0^A f^n dh \to -1/f'(0)$. Also $n\int_A^1 f^n dh \to 0$ since $f(A) < 1$. Thus $L = -1/f'(0)$. Analogous if $M = f(1)$, giving $L = 1/f'(1)$.

\section*{Problem 6 (Day 1)}
Upper content of $E \subset \mathbb{R}^2$:
\[
\mathcal{C}(E) = \inf\Big\{\sum_{i=1}^n \mathrm{diam}(E_i) : E \subset \bigcup E_i\Big\}.
\]
Lower content: $\mathcal{K}(E) = \sup\{\ell(L) : L \text{ closed segment, } E \text{ contracts onto } L\}$.

Show: (a) $\mathcal{C}(L) = \ell(L)$ for a segment; (b) $\mathcal{C}(E) \geq \mathcal{K}(E)$; (c) equality need not hold for compact $E$.

\subsection*{Solution}
(a) Choose $E_1 = L$, so $\mathcal{C}(L) \leq \ell(L)$. Conversely, for a cover $L \subset \bigcup E_i$, induct: pick $E_{n+1}$ containing endpoint $a$; the remaining $n$ sets cover $L$ except a portion of length $\leq \mathrm{diam}(E_{n+1}) + \varepsilon$.

(b) If $f: E \to L$ is a contraction onto $L$ and $E \subset \bigcup E_i$, then $L \subset \bigcup f(E_i)$ and $\ell(L) \leq \sum \mathrm{diam}(f(E_i)) \leq \sum \mathrm{diam}(E_i)$.

(c) Let $E = T \cup T'$ where $T$ is the triangle with vertices $(-2,2), (2,2), (0,4)$ and $T'$ its reflection about the $x$-axis.

(c1) $\mathcal{C}(E) \geq 8$: if no $E_i$ meets both $T$ and $T'$, the cover splits into covers of segments $[(-2,2),(2,2)]$ and $[(-2,-2),(2,-2)]$, each length $4$, so $\sum \mathrm{diam} \geq 8$. Otherwise some $E_k$ meets both; replacing by $E_k \cup [a,b]$ (for $a\in E_k\cap T, b\in E_k\cap T'$) gives a cover of $T\cup T'\cup[a,b]$, whose vertical projection has length $8$, so $\sum\mathrm{diam}\geq 8$.

(c2) $\mathcal{K}(E) < 8$: let $f$ be a contraction of $E$ onto $L=[a',b']$ with $f(a)=a'$, $f(b)=b'$. WLOG $a \in T$. If $a_2 \leq 3$, then $a$ is within distance $\sqrt{50}$ of every point of $E$, so $\ell(L) \leq \sqrt{50}$. If $a_2 > 3$ and $b_2 < -3$, every vertex of $T$ is within $\sqrt{10}$ of $a$, every vertex of $T'$ within $\sqrt{10}$ of $b$; then images of $T, T'$ lie in segments of length $\leq \sqrt{10}$ containing $a', b'$, so $\ell(L) \leq 2\sqrt{10} < 8$.

\section*{Problem 1 (Day 2)}
Prove: if $f:[0,1]\to[0,1]$ is continuous, the iterates $x_{n+1} = f(x_n)$ converge iff $\lim (x_{n+1} - x_n) = 0$.

\subsection*{Solution}
``Only if'' is immediate. Conversely, assume $x_{n+1}-x_n \to 0$ but $\{x_n\}$ diverges. Then there are cluster points $K < L$. Pick $x \in (K, L)$ with $f(x) \neq x$; let $\varepsilon = |f(x)-x|/2 > 0$. By continuity, $|f(y) - y| > \varepsilon$ for $y \in (x-\delta, x+\delta)$. For large $n$, $|x_{n+1}-x_n| = |f(x_n)-x_n| < \varepsilon$ and $|x_{n+1}-x_n| < 2\delta$, so $x_n \notin (x-\delta, x+\delta)$ yet cannot jump across it. Then all cluster points lie $\leq x - \delta$ or $\geq x + \delta$, contradicting the existence of $K$ or $L$.

\section*{Problem 2 (Day 2)}
Let $\theta > 0$, $\cosh t = (e^t + e^{-t})/2$. Show: if $k \in \mathbb{N}$ and $\cosh k\theta, \cosh(k+1)\theta$ are rational, so is $\cosh\theta$.

\subsection*{Solution}
First: if $\cosh t \in \mathbb{Q}$ and $m \in \mathbb{N}$, then $\cosh mt \in \mathbb{Q}$ â€” induction via
\[
\cosh(m+1)t = 2\cosh t \cosh mt - \cosh(m-1)t.
\]
Case $k=1$ is trivial; assume $k \geq 2$. For any $m$,
\[
\cosh\theta = \cosh(m+1)\theta\,\cosh m\theta - \sqrt{\cosh^2(m+1)\theta - 1}\sqrt{\cosh^2 m\theta - 1}.
\]
Set $a = \cosh k\theta, b = \cosh(k+1)\theta \in \mathbb{Q}$. With $m = k$:
\[
(a^2-1)(b^2-1) = (ab - \cosh\theta)^2 = a^2 b^2 - 2ab\cosh\theta + \cosh^2\theta. \quad (3)
\]
Set $A = \cosh(k^2-1)\theta$, $B = \cosh(k^2)\theta$; both rational (apply step one with $t=(k+1)\theta$, $m=k-1$ for $A$; $t=k\theta$, $m=k$ for $B$). Since $k^2 - 1 > k$, $AB > ab$. With $m = k^2 - 1$:
\[
(A^2-1)(B^2-1) = A^2 B^2 - 2AB\cosh\theta + \cosh^2\theta. \quad (4)
\]
Subtracting (3) from (4) cancels $\cosh^2\theta$ leaving a nontrivial linear equation for $\cosh\theta$ over $\mathbb{Q}$.

\section*{Problem 3 (Day 2)}
Let $G \leq GL_2(\mathbb{R})$ be generated by
\[
A = \begin{pmatrix}2 & 0\\0 & 1\end{pmatrix}, \quad B = \begin{pmatrix}1 & 1\\0 & 1\end{pmatrix}.
\]
Let $H = \{M \in G : M_{11} = M_{22} = 1\}$.
(a) Show $H$ is abelian.
(b) Show $H$ is not finitely generated.

\subsection*{Solution}
(a) All elements of $G$ are upper triangular $\begin{pmatrix}*&*\\0&*\end{pmatrix}$. Thus elements of $H$ have the form $M(x) = \begin{pmatrix}1 & x\\ 0 & 1\end{pmatrix}$; $M(x)M(y) = M(x+y)$, so $H$ is abelian, and $M(x)^{-1} = M(-x)$.

(b) A generator of $H$ is $M(x)$ with $x = p/2^n$ (dyadic rational) since all word-entries in $G$ have denominators that are powers of $2$. In $H$:
\[
M(x)M(y) = M(x+y), \quad M(x)M(y)^{-1} = M(x-y).
\]
The matrices $M(1/2^n)$ lie in $H$ for all $n$. With finitely many generators the denominators are bounded, contradicting the presence of $M(1/2^n)$ for arbitrarily large $n$.

\section*{Problem 4 (Day 2)}
Let $B \subset \mathbb{R}^2$ be closed bounded convex and symmetric with boundary $\Gamma$, such that the ellipse of maximal area in $B$ is the unit disc $D$ (boundary $C$). Prove $A \cap \Gamma \neq \emptyset$ for every arc $A \subset C$ with $\ell(A) \geq \pi/2$.

\subsection*{Solution}
Assume $A \subset B\setminus\Gamma$ with $\ell(A) = \pi/2$. WLOG endpoints $M = (1/\sqrt 2, 1/\sqrt 2)$, $N = (1/\sqrt 2, -1/\sqrt 2)$. Since $A$ compact and $\Gamma$ closed with $A\cap\Gamma=\emptyset$, $\exists\,\delta>0$ with $\mathrm{dist}(x,y) > \delta$ for $x\in A, y\in\Gamma$.

For $\varepsilon > 0$ consider ellipse $E_\varepsilon: x^2/(1+\varepsilon)^2 + y^2/b^2 = 1$ through $M, N$: $b^2 = (1+\varepsilon)^2 / (2(1+\varepsilon)^2 - 1)$. Its area is $\pi(1+\varepsilon)^2/\sqrt{2(1+\varepsilon)^2-1} > \pi = $ area$(D)$.

By maximality, $E_\varepsilon \not\subset B$. Let $S = \{(x,y): |x| > |y|\}$; then $E_\varepsilon \setminus S \subset D \subset B$, so $E_\varepsilon \setminus B \subset S$. For $\varepsilon < \delta$, $E_\varepsilon \cap S \subset D_{1+\varepsilon} \cap S$ is within $\varepsilon < \delta$ of $D$, i.e. $\subset B$. Contradiction.

\section*{Problem 5 (Day 2)}
(i) Prove $\lim_{x\to+\infty} \sum_{n=1}^\infty \frac{nx}{(n^2+x)^2} = \frac{1}{2}$.
(ii) Prove there is $c>0$ with, for $x \geq 1$,
\[
\Big|\sum_{n=1}^\infty \frac{nx}{(n^2+x)^2} - \tfrac{1}{2}\Big| \leq \frac{c}{x}.
\]

\subsection*{Solution}
(i) Set $f(t) = t/(1+t^2)^2$, $h = 1/\sqrt{x}$. Then
\[
\sum_{n=1}^\infty \frac{nx}{(n^2+x)^2} = h\sum_{n=1}^\infty f(nh) \xrightarrow{h\to 0} \int_0^\infty f(t)\,dt = \tfrac{1}{2}.
\]
Convergence holds since $f$ is integrable and $f\downarrow 0$ on $(1,\infty)$.

(ii) We have
\[
\Big|\sum hf(nh) - \int_0^\infty f\Big| \leq \sum_n\Big|hf(nh) - \int_{nh-h/2}^{nh+h/2} f\Big| + \int_0^{h/2} f.
\]
Using integration by parts twice, for $g\in C^2[a-b, a+b]$,
\[
2bg(a) - \int_{a-b}^{a+b} g(t)dt = -\tfrac{1}{2}\int_0^b (b-t)^2 (g''(a+t) + g''(a-t))dt.
\]
With $f(0) = 0$, $f \in C^2[0, h/2]$: $\int_0^{h/2} f\,dt = O(h^2)$. So
\[
\Big|\sum - \tfrac{1}{2}\Big| \leq h^2 \int_{h/2}^\infty |f''| + O(h^2) = O(h^2) = O(1/x).
\]

\section*{Problem 6 (Day 2): Carleman's inequality}
(i) For $a_n > 0$ with $\sum a_n < \infty$: $\sum (a_1 a_2 \cdots a_n)^{1/n} < e \sum a_n$.
(ii) For every $\varepsilon > 0$, $\exists$ a sequence with $\sum (a_1\cdots a_n)^{1/n} > (e-\varepsilon)\sum a_n$.

\subsection*{Solution}
(i) Put $c_n = (n+1)^n / n^{n-1}$. Then $c_1\cdots c_n = (n+1)^n$, so by AM-GM
\[
(a_1\cdots a_n)^{1/n} = \frac{(a_1 c_1\cdots a_n c_n)^{1/n}}{n+1} \leq \frac{a_1 c_1 + \cdots + a_n c_n}{n(n+1)}.
\]
Summing and interchanging order:
\[
\sum_{n=1}^\infty (a_1\cdots a_n)^{1/n} \leq \sum_{n=1}^\infty a_n c_n \sum_{m=n}^\infty \tfrac{1}{m(m+1)} = \sum_{n=1}^\infty \tfrac{a_n c_n}{n} = \sum_{n=1}^\infty a_n\Big(\tfrac{n+1}{n}\Big)^n < e\sum a_n.
\]

(ii) Set $a_n = n^{n-1}(n+1)^{-n}$ for $n\leq N$, $a_n = 2^{-n}$ for $n > N$ ($N$ chosen later). Then $(a_1\cdots a_n)^{1/n} = 1/(n+1)$ for $n\leq N$. Fix $K$ with $(1+1/n)^n > e - \varepsilon/2$ for $n > K$; choose $N$ so
\[
\sum_{n=1}^K a_n + \sum_{n=1}^\infty 2^{-n} \leq \tfrac{\varepsilon}{(2e-\varepsilon)(e-\varepsilon)} \sum_{n=K+1}^N \tfrac{1}{n},
\]
possible since the harmonic series diverges. Then
\[
\sum a_n < \tfrac{1}{e-\varepsilon}\sum_{n=K+1}^N \tfrac{1}{n} \leq \tfrac{1}{e-\varepsilon}\sum (a_1\cdots a_n)^{1/n}.
\]
