\section*{Problem 6}
Let $f, g: \mathbb{R} \to \mathbb{R}$ be continuous with $g$ differentiable. Assume $(f(0) - g'(0))(g'(1) - f(1)) > 0$. Show there exists $c \in (0, 1)$ with $f(c) = g'(c)$.

\section*{Problem 7}
Let $C = \{4, 6, 8, 9, 10, \ldots\}$ be the composite positive integers. For $n \in C$ let $a_n$ be the smallest $k$ such that $n \mid k!$. Determine if $\sum_{n \in C} (a_n/n)^n$ converges.

\section*{Problem 8}
Let $x_1, \ldots, x_n \in \mathbb{R}$. For $I \subseteq \{1, \ldots, n\}$, let $s(I) = \sum_{i \in I} x_i$. If $I \mapsto s(I)$ takes at least $1.8^n$ values, prove that at most $1.7^n$ sets $I$ satisfy $s(I) = 2019$.

\section*{Problem 9}
Determine all positive integers $n$ for which there exist invertible real $n \times n$ matrices $A, B$ satisfying $AB - BA = B^2 A$.

\section*{Problem 10}
$2019$ points are chosen uniformly and independently in the unit disc. Let $C$ be their convex hull. Which is larger: $\mathbb{P}(C \text{ is a triangle})$ or $\mathbb{P}(C \text{ is a quadrilateral})$?
