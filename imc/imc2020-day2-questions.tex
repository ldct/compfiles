\section*{Problem 5}
Find all twice continuously differentiable functions $f : \mathbb{R} \to (0, +\infty)$ satisfying
\[ f''(x) f(x) \geq 2 (f'(x))^2 \quad \text{for all } x \in \mathbb{R}. \]

\section*{Problem 6}
Find all prime numbers $p$ for which there exists a unique $a \in \{1, 2, \ldots, p\}$ such that $a^3 - 3a + 1$ is divisible by $p$.

\section*{Problem 7}
Let $G$ be a group and $n \geq 2$ an integer. Let $H_1, H_2$ be subgroups of $G$ satisfying
\[ [G : H_1] = [G : H_2] = n \quad \text{and} \quad [G : H_1 \cap H_2] = n(n-1). \]
Prove that $H_1$ and $H_2$ are conjugate in $G$.

\section*{Problem 8}
Compute
\[ \lim_{n \to \infty} \frac{1}{\log \log n} \sum_{k=1}^n (-1)^k \binom{n}{k} \log k. \]
