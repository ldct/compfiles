\section*{Problem 1}
(a) Show for any $m\in\mathbb{N}$ there is a real $m\times m$ matrix $A$ with $A^3 = A+I$.

(b) Show $\det A > 0$ for any such matrix.

\subsection*{Solution}
(a) Take $A = \lambda I$ where $\lambda$ is a real root of $x^3-x-1=0$; such $\lambda$ exists since $p(0)=-1<0$.

(b) $p(x)=x^3-x-1$ has one real root $\lambda_1>0$ and two complex conjugate roots $\lambda_2, \lambda_3$ (discriminant $(-1/3)^3+(-1/2)^2 = 23/108 > 0$). Eigenvalues of $A$ are among these. Since $A$ is real, $\lambda_2, \lambda_3$ have equal multiplicity $\beta$. Let $\lambda_1$ have multiplicity $\alpha$:
\[\det A = \lambda_1^\alpha (\lambda_2\lambda_3)^\beta = \lambda_1^\alpha |\lambda_2|^{2\beta} > 0.\]

\section*{Problem 2}
Does there exist a bijection $\pi:\mathbb{N}\to\mathbb{N}$ with $\sum_{n=1}^\infty \pi(n)/n^2 < \infty$?

\subsection*{Solution}
No. For any $N$, among $\pi(N+1),\dots,\pi(3N)$ at most $N$ can be $\le N$, so at least $N$ exceed $N$. Hence
\[\sum_{n=N+1}^{3N}\frac{\pi(n)}{n^2} \ge \frac{1}{(3N)^2}\sum_{n=N+1}^{3N}\pi(n) > \frac{1}{9N^2}\cdot N\cdot N = \frac 19.\]
Taking $N\to\infty$ shows divergence.

\section*{Problem 3}
Suppose $f:\mathbb{R}\to\mathbb{R}$ satisfies $\left|\sum_{k=1}^n 3^k(f(x+ky)-f(x-ky))\right|\le 1$ for all $n,x,y$. Prove $f$ is constant.

\subsection*{Solution}
Subtracting the $n-1$ inequality from the $n$ inequality: $|3^n(f(x+ny)-f(x-ny))|\le 2$, so $|f(x+ny)-f(x-ny)|\le 2/3^n$.

For any $u<v$, pick $x=(u+v)/2$, $y=(v-u)/(2n)$ to get $|f(u)-f(v)|\le 2/3^n$. Letting $n\to\infty$, $f(u)=f(v)$.

\section*{Problem 4}
Find all strictly monotonic $f:(0,\infty)\to(0,\infty)$ with $f(x^2/f(x)) = x$.

\subsection*{Solution}
Let $g(x)=f(x)/x$. The equation gives $g(x/g(x)) = g(x)$, and by induction $f(x/g^n(x)) = x/g^{n-1}(x)$. Substituting $x\to f(x)$ and using injectivity yields $g(xg(x))=g(x)$, so $f(xg^n(x))=xg^{n-1}(x)$.

Combining, $f^{(m)}(xg^n(x)) = xg^{n-m}(x)$. If $g(x_1)<g(x_2)$, pick $n$ with $x_1 g^n(x_1)\le x_2 g^n(x_2)$. For $m$ even, $f^{(m)}$ is increasing, so $x_1 g^{n-m}(x_1)\le x_2 g^{n-m}(x_2)$; but for large $m$ this fails (reverse inequality). Contradiction: $g$ is constant.

So $f(x) = Cx$ with $C>0$, and the original equation forces $C^2 \cdot (x/C) = x$, so $C = C$ is arbitrary positive. Verifying: $f(x^2/f(x)) = f(x/C) = x$. Answer: $f(x)=Cx$, $C>0$.

\section*{Problem 5}
If $2n$ points are marked on an $n\times n$ grid, show that for some $k>1$ one can pick $2k$ distinct marked points $a_1,\dots,a_{2k}$ with $a_1,a_2$ in a row, $a_2,a_3$ in a column, $\dots$, $a_{2k-1},a_{2k}$ in a row, $a_{2k},a_1$ in a column.

\subsection*{Solution}
Make a graph $G$ with vertices = rows $\cup$ columns, and edge between $r$ and $c$ iff $(r,c)$ is marked. Then $G$ has $2n$ vertices, $2n$ edges, so $G$ contains a cycle. A cycle alternates rows and columns, and its consecutive edges correspond to marked points giving the required sequence.

\section*{Problem 6}
(a) For $1<p<\infty$ find $c_p<\infty$ such that if $f\in C^1[-1,1]$ with $f(1)>f(-1)$ and $|f'|\le 1$, then there is $x\in[-1,1]$ with $f'(x)>0$ and $|f(y)-f(x)|\le c_p (f'(x))^{1/p}|y-x|$ for all $y$.

(b) Does $c_1$ exist?

\subsection*{Solution}
(a) Let $g(x)=\max(0,f'(x))$. Then $\int|f'|\le 2\int g$. Suppose the claim fails for $c=c_p$. For each $t>0$ and each $x$ with $g(x)>t$, take an interval $I_x=[x,y]$ with $|f(y)-f(x)|>c t^{1/p}|I_x|$. Vitali covering gives disjoint $I_{x_i}$ covering at least $\frac 13$ of $\{g>t\}$. Then $ct^{1/p}|I| \le \int_I f' \le 2\int g$, so $|\{g>t\}|\le (6/c)t^{-1/p}\int g$. Integrating in $t$:
\[\int g = \int_0^1 |\{g>t\}|\,dt \le (6/c)\cdot p/(p-1)\cdot \int g,\]
contradiction for $c_p = 6p/(p-1)$.

(b) No. Given $c>1$, $\alpha=1/c$, pick $0<\varepsilon<1$ with $((1+\varepsilon)/(2\varepsilon))^{-\alpha}<1/4$. Let $g$ be even continuous on $[-1,1]$, $g(x)=-1$ for $|x|\le\varepsilon$, $0\le g(x)<\alpha((|x|+\varepsilon)/(2\varepsilon))^{-\alpha-1}$ for $\varepsilon<|x|\le 1$, chosen with $\int_\varepsilon^1 g > \varepsilon$. Let $f=\int g$. Then $f(1)-f(-1)>0$. For $\varepsilon<x<1$ and $y=-\varepsilon$, $|f(x)-f(y)|\ge 2\varepsilon((x+\varepsilon)/(2\varepsilon))^{-\alpha} > g(x)|x-y|/\alpha = f'(x)|x-y|/\alpha$. No valid $x$ exists.
