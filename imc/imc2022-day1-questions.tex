\section*{Problem 1}
Let $f : [0,1] \to (0, \infty)$ be an integrable function such that $f(x) \cdot f(1-x) = 1$ for all $x \in [0,1]$. Prove that
\[ \int_0^1 f(x)\, dx \geq 1. \]

\section*{Problem 2}
Let $n$ be a positive integer. Find all $n \times n$ real matrices $A$ with only real eigenvalues satisfying
\[ A + A^k = A^T \]
for some integer $k \geq n$. ($A^T$ denotes the transpose of $A$.)

\section*{Problem 3}
Let $p$ be a prime number. A flea is staying at point $0$ of the real line. At each minute, the flea has three possibilities: to stay at its position, or to move by $1$ to the left or to the right. After $p-1$ minutes, it wants to be at $0$ again. Denote by $f(p)$ the number of its strategies to do this (for example, $f(3) = 3$). Find $f(p)$ modulo $p$.

\section*{Problem 4}
Let $n > 3$ be an integer. Let $\Omega$ be the set of all triples of distinct elements of $\{1, 2, \ldots, n\}$. Let $m$ denote the minimal number of colours which suffice to colour $\Omega$ so that whenever $1 \leq a < b < c < d \leq n$, the triples $\{a,b,c\}$ and $\{b,c,d\}$ have different colours. Prove that
\[ \tfrac{1}{100} \log \log n \leq m \leq 100 \log \log n. \]
