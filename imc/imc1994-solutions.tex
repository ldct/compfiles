\section*{Problem 1 (Day 1)}
(a) Let $A$ be an $n\times n$ ($n\geq 2$) symmetric, invertible matrix with positive real entries. Show $z_n \leq n^2 - 2n$, where $z_n$ is the number of zero entries in $A^{-1}$.

(b) How many zero entries are in the inverse of the $n\times n$ matrix
\[
A = \begin{pmatrix}
1 & 1 & 1 & 1 & \cdots & 1\\
1 & 2 & 2 & 2 & \cdots & 2\\
1 & 2 & 1 & 1 & \cdots & 1\\
1 & 2 & 1 & 2 & \cdots & 2\\
\vdots & & & & & \vdots\\
1 & 2 & 1 & 2 & \cdots & \cdots
\end{pmatrix}?
\]

\subsection*{Solution}
Let $a_{ij}, b_{ij}$ be entries of $A, A^{-1}$. For $k \neq m$, $\sum_{i=0}^n a_{ki} b_{im} = 0$. Since $a_{ki} > 0$, $\{b_{im}\}_i$ must have at least one positive and at least one negative entry, so at least two nonzero entries per column of $A^{-1}$. Hence at most $n^2 - 2n$ zeros.

(b) All $b_{ij}$ are zero except $b_{1,1} = 2$, $b_{n,n} = (-1)^n$, $b_{i,i+1} = b_{i+1,i} = (-1)^i$ for $i = 1,\ldots,n-1$. So $n^2 - 2n$ zeros.

\section*{Problem 2 (Day 1)}
Let $f \in C^1(a,b)$ with $\lim_{x\to a+} f(x) = +\infty$, $\lim_{x\to b-}f(x) = -\infty$ and $f'(x) + f^2(x) \geq -1$ on $(a,b)$. Prove $b - a \geq \pi$ and give an example where $b - a = \pi$.

\subsection*{Solution}
From the inequality,
\[
\frac{d}{dx}(\arctan f(x) + x) = \frac{f'(x)}{1 + f^2(x)} + 1 \geq 0,
\]
so $\arctan f(x) + x$ is nondecreasing. Taking limits: $\pi/2 + a \leq -\pi/2 + b$, giving $b - a \geq \pi$. Equality: $f(x) = \cot x$ on $(0, \pi)$.

\section*{Problem 3 (Day 1)}
Given a set $S$ of $2n - 1$ distinct irrational numbers, prove there exist $n$ distinct elements $x_1, \ldots, x_n \in S$ such that for all nonnegative rationals $a_1, \ldots, a_n$ with $a_1 + \cdots + a_n > 0$, the sum $a_1 x_1 + \cdots + a_n x_n$ is irrational.

\subsection*{Solution}
Let $\mathbb{I}$ denote irrationals, $\mathbb{Q}^+ = \mathbb{Q} \cap [0, \infty)$. Induct on $n$. For $n=1$ trivial. Assume true for $n-1$: there exist $x_1, \ldots, x_{n-1} \in S$ with
\[
a_1 x_1 + \cdots + a_{n-1}x_{n-1} \in \mathbb{I} \quad \forall a_i \in \mathbb{Q}^+, \sum a_i > 0. \quad (1)
\]
Denote the remaining elements of $S$ by $x_n, x_{n+1}, \ldots, x_{2n-1}$. Suppose the claim fails for $n$. Then for each $k = 0, 1, \ldots, n-1$, there are $b_{ik}, c_k \in \mathbb{Q}^+$ not all zero with
\[
\sum_{i=1}^{n-1} b_{ik} x_i + c_k x_{n+k} = r_k \in \mathbb{Q}, \quad \sum_i b_{ik} + c_k > 0. \quad (2)
\]
Also (applying (1) in reverse) there exist $d_k \in \mathbb{Q}^+$, $\sum d_k > 0$, with $\sum_{k=0}^{n-1} d_k x_{n+k} = R \in \mathbb{Q}$. (3)

If $c_k = 0$, (2) contradicts (1). So $c_k \neq 0$; WLOG $c_k = 1$, and $\sum_i b_{ik} > 0$ (since $x_{n+k}$ irrational). Substitute (2) into (3):
\[
\sum_{k=0}^{n-1} d_k\Big(-\sum_i b_{ik} x_i + r_k\Big) = R \implies \sum_{i=1}^{n-1}\Big(\sum_k d_k b_{ik}\Big) x_i \in \mathbb{Q},
\]
contradicting (1) since $\sum_k d_k b_{ik} \geq 0$ with some strictly positive.

\section*{Problem 4 (Day 1)}
Let $\alpha \in \mathbb{R}\setminus\{0\}$ and $F, G: \mathbb{R}^n \to \mathbb{R}^n$ linear with $F\circ G - G\circ F = \alpha F$.
(a) Show $F^k \circ G - G \circ F^k = \alpha k F^k$ for all $k$.
(b) Show there exists $k \geq 1$ with $F^k = 0$.

\subsection*{Solution}
(a) Telescoping:
\[
F^k \circ G - G \circ F^k = \sum_{i=1}^k (F^{k-i+1} \circ G \circ F^{i-1} - F^{k-i} \circ G \circ F^i) = \sum_{i=1}^k F^{k-i} \circ (F\circ G - G\circ F) \circ F^{i-1} = \alpha k F^k.
\]

(b) The operator $L(F) = F\circ G - G\circ F$ on $n\times n$ matrices has at most $n^2$ distinct eigenvalues. If $F^k \neq 0$ for all $k$, then by (a) each $\alpha k$ is an eigenvalue — infinitely many, contradiction.

\section*{Problem 5 (Day 1)}
(a) Let $f \in C[0,b]$, $g \in C(\mathbb{R})$ periodic with period $b$. Prove
\[
\lim_{n\to\infty}\int_0^b f(x)g(nx)\,dx = \frac{1}{b}\int_0^b f(x)dx \cdot \int_0^b g(x)dx.
\]
(b) Find $\lim_{n\to\infty}\int_0^\pi \frac{\sin x}{1 + 3\cos^2 nx}\,dx$.

\subsection*{Solution}
Let $\|g\|_1 = \int_0^b |g|$ and $\omega(f,t) = \sup\{|f(x)-f(y)| : |x-y|\leq t\}$; $\omega(f,t)\to 0$ by uniform continuity. Split into intervals $[b(k-1)/n, bk/n]$ and use periodicity:
\[
\int_0^b f(x)g(nx)dx = \sum_{k=1}^n f(bk/n)\int_{b(k-1)/n}^{bk/n} g(nx)dx + O(\omega(f,b/n)\|g\|_1).
\]
Each $\int_{b(k-1)/n}^{bk/n}g(nx)dx = \frac{1}{n}\int_0^b g$. Hence the sum is $\frac{1}{n}\sum f(bk/n) \int_0^b g$, a Riemann sum equal to $\frac{1}{b}\int_0^b f \cdot \int_0^b g + O(\omega(f,b/n)\|g\|_1)$.

(b) $b = \pi$, $f = \sin x$, $g(x) = 1/(1 + 3\cos^2 x)$. Then $\int_0^\pi \sin x\,dx = 2$ and $\int_0^\pi dx/(1+3\cos^2 x) = \pi/2$, so
\[
\lim_{n\to\infty}\int_0^\pi \frac{\sin x}{1+3\cos^2 nx}\,dx = \frac{1}{\pi} \cdot 2 \cdot \frac{\pi}{2} = 1.
\]

\section*{Problem 6 (Day 1)}
Let $f \in C^2[0, N]$ with $|f'(x)| < 1$ and $f''(x) > 0$ on $[0, N]$. Let $0 \leq m_0 < m_1 < \cdots < m_k \leq N$ be integers with $n_i = f(m_i)$ also integers. Let $b_i = n_i - n_{i-1}$, $a_i = m_i - m_{i-1}$.
(a) Prove $-1 < b_1/a_1 < b_2/a_2 < \cdots < b_k/a_k < 1$.
(b) For $A > 1$, show there are at most $N/A$ indices $j$ with $a_j > A$.
(c) Show $k \leq 3 N^{2/3}$.

\subsection*{Solution}
(a) By MVT: $b_i = f(m_i) - f(m_{i-1}) = a_i f'(x_i)$ for some $x_i \in (m_{i-1}, m_i)$. So $b_i/a_i = f'(x_i)$, giving $|b_i/a_i| < 1$. Convexity $\Rightarrow f'$ increasing, and $x_i < m_i < x_{i+1}$ gives $b_i/a_i < b_{i+1}/a_{i+1}$.

(b) Let $S_A = \{j : a_j > A\}$. Then
\[
N \geq m_k - m_0 = \sum_{i=1}^k a_i \geq \sum_{j \in S_A} a_j > A|S_A|,
\]
so $|S_A| < N/A$.

(c) The values $b_i/a_i$ are distinct rationals in $(-1,1)$. The number of such rationals with denominator $\leq A$ is at most $2A^2$. Using (b), $k \leq N/A + 2A^2$. Setting $A = N^{1/3}$ gives $k \leq 3N^{2/3}$.

\section*{Problem 1 (Day 2)}
Let $f \in C^1[a,b]$, $f(a) = 0$, and $\lambda > 0$ such that $|f'(x)| \leq \lambda |f(x)|$ on $[a,b]$. Is $f(x) = 0$ on $[a,b]$?

\subsection*{Solution}
Yes. Suppose $f(y) \neq 0$ for some $y \in (a,b]$; WLOG $f(y) > 0$. By continuity, $\exists\, c \in [a, y)$ with $f(c) = 0$ and $f > 0$ on $(c, y]$. On $(c, y]$, $|f'(x)| \leq \lambda f(x)$, so $g(x) = \ln f(x) - \lambda x$ satisfies $g'(x) = f'(x)/f(x) - \lambda \leq 0$, i.e., $g$ is non-increasing. Hence $\ln f(x) - \lambda x \geq \ln f(y) - \lambda y$, so $f(x) \geq e^{\lambda x - \lambda y} f(y)$ on $(c, y]$. Then
\[
0 = f(c) = \lim_{x\to c+} f(x) \geq e^{\lambda c - \lambda y} f(y) > 0,
\]
contradiction.

\section*{Problem 2 (Day 2)}
Let $f(x,y) = (x^2 - y^2)e^{-x^2 - y^2}$.
(a) Prove $f$ attains its minimum and maximum.
(b) Determine all critical points and classify.

\subsection*{Solution}
$f(1,0) = e^{-1}$, $f(0,1) = -e^{-1}$, and $te^{-t} \leq 2e^{-2}$ for $t \geq 2$; so $|f(x,y)| \leq (x^2+y^2)e^{-x^2-y^2} \leq 2e^{-2} < e^{-1}$ for $(x,y) \notin M = \{(u,v):u^2+v^2\leq 2\}$. Thus extrema are inside $M$; compactness gives (a).

Critical points from $\partial f/\partial x = 2x(1 - x^2 + y^2)e^{-x^2-y^2} = 0$ and $\partial f/\partial y = -2y(1 + x^2 - y^2)e^{-x^2-y^2} = 0$:
\[
x(1 - x^2 + y^2) = 0, \quad y(1 + x^2 - y^2) = 0.
\]
Solutions: $(0,0), (0,1), (0,-1), (1,0), (-1,0)$. $f(\pm 1, 0) = e^{-1}$ (global max), $f(0, \pm 1) = -e^{-1}$ (global min). At $(0,0)$: $f(x,0) = x^2 e^{-x^2} > 0$ for $x \neq 0$, $f(0,y) = -y^2 e^{-y^2} < 0$ for $y \neq 0$ — saddle.

\section*{Problem 3 (Day 2)}
Let $f$ have $n+1$ derivatives on $\mathbb{R}$. Show: for each $a < b$ with
\[
\ln\left(\frac{f(b) + f'(b) + \cdots + f^{(n)}(b)}{f(a) + f'(a) + \cdots + f^{(n)}(a)}\right) = b - a,
\]
there is $c \in (a,b)$ with $f^{(n+1)}(c) = f(c)$.

\subsection*{Solution}
Set $g(x) = (f(x) + f'(x) + \cdots + f^{(n)}(x))e^{-x}$. The hypothesis gives $g(a) = g(b)$, so by Rolle there is $c \in (a,b)$ with $g'(c) = 0$. But
\[
g'(x) = (f^{(n+1)}(x) - f(x))e^{-x},
\]
so $f^{(n+1)}(c) = f(c)$.

\section*{Problem 4 (Day 2)}
Let $A$ be an $n\times n$ diagonal matrix with characteristic polynomial $(x-c_1)^{d_1}\cdots(x-c_k)^{d_k}$ ($c_i$ distinct, $\sum d_i = n$). Let $V = \{B : AB = BA\}$. Prove $\dim V = d_1^2 + \cdots + d_k^2$.

\subsection*{Solution}
With $A = (a_{ij})$ diagonal and $B = (b_{ij})$, $(AB)_{ij} = a_{ii} b_{ij}$ and $(BA)_{ij} = a_{jj} b_{ij}$. So $AB = BA$ iff $(a_{ii} - a_{jj})b_{ij} = 0$ for all $i,j$. Thus $b_{ij} = 0$ unless $a_{ii} = a_{jj}$, and $b_{ij}$ is free otherwise. The number of $(i,j)$ with $a_{ii} = a_{jj} = c_m$ is $d_m^2$. Hence $\dim V = \sum d_m^2$.

\section*{Problem 5 (Day 2)}
Let $x_1, \ldots, x_k \in \mathbb{R}^m$ with $x_1 + \cdots + x_k = 0$. Show there is a permutation $\pi$ with
\[
\Big\|\sum_{i=1}^n x_{\pi(i)}\Big\| \leq \Big(\sum_{i=1}^k \|x_i\|^2\Big)^{1/2} \quad \forall n = 1, \ldots, k.
\]

\subsection*{Solution}
Define $\pi$ inductively: $\pi(1) = 1$. Suppose $\pi(1), \ldots, \pi(n)$ defined with
\[
\Big\|\sum_{i=1}^n x_{\pi(i)}\Big\|^2 \leq \sum_{i=1}^n \|x_{\pi(i)}\|^2. \quad (1)
\]
Let $y = \sum_{i=1}^n x_{\pi(i)}$ and $A = \{1,\ldots,k\}\setminus\{\pi(i)\}$. Then $\sum_{r\in A} x_r = -y$. If $(y, x_r) > 0$ for all $r \in A$, summing gives $(y, -y) > 0$, i.e. $-\|y\|^2 > 0$, impossible. Pick $r \in A$ with $(y, x_r) \leq 0$; set $\pi(n+1) = r$:
\[
\Big\|\sum_{i=1}^{n+1} x_{\pi(i)}\Big\|^2 = \|y\|^2 + 2(y, x_r) + \|x_r\|^2 \leq \|y\|^2 + \|x_r\|^2 \leq \sum_{i=1}^{n+1} \|x_{\pi(i)}\|^2.
\]
By induction $(1)$ holds for all $n$, and $\sum \|x_{\pi(i)}\|^2 \leq \sum_{i=1}^k \|x_i\|^2$.

\section*{Problem 6 (Day 2)}
Find
\[
\lim_{N\to\infty}\frac{\ln^2 N}{N}\sum_{k=2}^{N-2}\frac{1}{\ln k \cdot \ln(N-k)}.
\]

\subsection*{Solution}
Let $A_N$ be the sum. Lower bound: $\ln k \cdot \ln(N-k) \leq \ln^2 N$, so
\[
A_N \geq \frac{\ln^2 N}{N} \cdot \frac{N-3}{\ln^2 N} = 1 - 3/N. \quad (1)
\]

Upper bound: $1/(\ln k \cdot \ln(N-k))$ is decreasing on $[2, N/2]$ and symmetric about $N/2$. For $2 \leq M < N/2$, split $\sum_{k=2}^{N-2} = \sum_{k=2}^M + \sum_{k=M+1}^{N-M-1} + \sum_{k=N-M}^{N-2}$:
\[
A_N \leq \frac{\ln^2 N}{N}\Big\{2\frac{M-1}{\ln 2 \cdot \ln(N-2)} + \frac{N-2M-1}{\ln M \cdot \ln(N-M)}\Big\}
\leq \frac{2M\ln N}{N\ln 2} + \Big(1 - \frac{2M}{N}\Big)\frac{\ln N}{\ln M} + O(1/\ln N).
\]
Choose $M = \lfloor N/\ln^2 N\rfloor + 1$:
\[
A_N \leq \Big(1 - \frac{2}{N\ln^2 N}\Big)\frac{\ln N}{\ln N - 2\ln\ln N} + O(1/\ln N) \leq 1 + O(\ln\ln N / \ln N). \quad (2)
\]

From (1), (2):
\[
\lim_{N\to\infty}\frac{\ln^2 N}{N}\sum_{k=2}^{N-2}\frac{1}{\ln k \cdot \ln(N-k)} = 1.
\]
