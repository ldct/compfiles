\section*{Problem 1}
Let $n$ be a positive integer. Compute the number of words $w$ (finite sequences of letters) that satisfy:
\begin{itemize}
\item[(1)] $w$ consists of $n$ letters from $\{a, b, c, d\}$;
\item[(2)] $w$ contains an even number of letters $a$;
\item[(3)] $w$ contains an even number of letters $b$.
\end{itemize}

\section*{Problem 2}
Let $A$ and $B$ be $n \times n$ real matrices such that $\operatorname{rk}(AB - BA + I) = 1$. Prove that
\[ \operatorname{trace}(ABAB) - \operatorname{trace}(A^2 B^2) = \tfrac{1}{2} n(n-1). \]

\section*{Problem 3}
Let $d \geq 2$ be an integer. Prove that there exists a constant $C(d)$ such that for any centrally symmetric convex polytope $K \subset \mathbb{R}^d$ and any $\varepsilon \in (0, 1)$, there exists a convex polytope $L \subset \mathbb{R}^d$ with at most $C(d) \varepsilon^{1-d}$ vertices such that $(1 - \varepsilon) K \subseteq L \subseteq K$.

\section*{Problem 4}
A polynomial $p$ with real coefficients satisfies $p(x+1) - p(x) = x^{100}$ for all $x \in \mathbb{R}$. Prove that $p(1-t) \geq p(t)$ for $0 \leq t \leq 1/2$.
