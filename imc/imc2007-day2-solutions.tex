\section*{Problem 1}
Let $f : \mathbb{R} \to \mathbb{R}$ be continuous. Suppose that for every $c > 0$, the graph of $f$ can be moved to the graph of $cf$ by a translation or rotation. Does $f(x) = ax + b$?

\subsection*{Solution}
\textbf{No.} $f(x) = e^x$ satisfies $c e^x = e^{x + \log c}$, a translation.

\section*{Problem 2}
Let $x, y, z \in \mathbb{Z}$ with $29 \mid x^4 + y^4 + z^4$. Show $29^4 \mid S$.

\subsection*{Solution}
It suffices to show $29 \mid x, y, z$.

Suppose $29 \nmid x$. Multiply by $w^4$ where $xw \equiv 1$: WLOG $x \equiv 1$. Need $y^4 + z^4 \equiv -1 \pmod{29}$.

Fourth powers mod 29 are: $\{0, 1, 7, 16, 20, 23, 24, 25\}$ (using $29 = 4 \cdot 7 + 1$, exponent $4$ gives $\tfrac{28}{\gcd(28,4)} = 7$ nonzero values).

$-1 - z^4$ for these $z^4$: $28, 27, 21, 12, 8, 5, 4, 3$. None lies in the fourth-power set. So no solution, contradiction.

\section*{Problem 3}
$C \subset \mathbb{R}$ nonempty closed bounded, $f : C \to C$ continuous nondecreasing. Show $\exists p \in C$ with $f(p) = p$.

\subsection*{Solution}
Suppose $f(x) \ne x$ on $C$. Let $[a, b]$ be the smallest closed interval containing $C$; $a, b \in C$. Since $a \le f(a) \in C \subseteq [a, b]$ and $f(a) \ne a$, $f(a) > a$. Similarly $f(b) < b$. Let $p = \sup\{x \in C : f(x) > x\}$; by closedness and continuity, $f(p) \ge p$, so $f(p) > p$. For all $x > p, x \in C$, we have $f(x) < x$. Since $f(p) \in C$ and $f(p) > p$, apply: $f(f(p)) < f(p)$, contradicting $f$ nondecreasing (as $p < f(p)$ should give $f(p) \le f(f(p))$).

\section*{Problem 4}
$n > 1$ odd; $A = (a_{ij})$ with $a_{ii} = 2$, $a_{ij} = 1$ if $i - j \equiv \pm 2 \pmod n$, else $0$. Find $\det A$.

\subsection*{Solution}
Note $A = B^2$ where $B$ has $b_{ij} = 1$ if $i - j \equiv \pm 1 \pmod n$ (so $B$ is the adjacency matrix of the $n$-cycle).

For $n$ odd, $\det B = 2$: expand along first row and then columns; ultimately reduces to $-(0 - 1) + (1 - 0) = 2$, as shown by the computation.

Therefore $\det A = (\det B)^2 = 4$.

\section*{Problem 5}
Find smallest $n_k$ so there exist $n_k \times n_k$ real matrices $A_1, \ldots, A_k$ with $A_i^2 = 0$, $A_i A_j = A_j A_i$, and $A_1 \cdots A_k \ne 0$.

\subsection*{Solution}
$n_k = 2^k$.

\emph{Construction:} $V$ has basis $\{[S] : S \subseteq \{1, \ldots, k\}\}$, dimension $2^k$. $A_i[S] = [S \cup \{i\}]$ if $i \notin S$, else $0$. Then $A_i^2 = 0$, $A_i A_j = A_j A_i$, $A_1 \cdots A_k [\emptyset] = [\{1, \ldots, k\}] \ne 0$.

\emph{Lower bound:} Pick $v$ with $A_1 \cdots A_k v \ne 0$. For $X \subseteq \{1, \ldots, k\}$, let $A_X = \prod_{i \in X} A_i$ and $v_X = A_X v$. Claim: $\{v_X : X \subseteq \{1, \ldots, k\}\}$ is linearly independent.

Order $\mathcal P$ by a total order $\prec$ refining size. For $X \subsetneq Y$, $A_{\bar X} v_Y = 0$ (some $y \in Y \setminus X$; $A_y^2 = 0$). So $A_{\bar X}$ annihilates $\operatorname{span}\{v_Y : X \subsetneq Y\}$. But $A_{\bar X} v_X = v_{\{1, \ldots, k\}} \ne 0$. By induction on $\prec$, $\{v_X\}$ are independent. Hence $n \ge 2^k$.

\section*{Problem 6}
$f \ne 0$ real polynomial; $f_0 = f$, $f_{n+1} = f_n + f_n'$. Prove for large $n$, $f_n$ has all real roots.

\subsection*{Solution}
\textbf{Lemma 1.} If $g$ and $g + g'$ both have $\deg = k \ge 2$ with $k$ distinct real roots, then $d(g + g') \ge d(g)$ (minimum root distance).

\emph{Proof.} Let $x_1 < \cdots < x_k$ be roots of $g$. Suppose $a, b$ are roots of $g + g'$ with $0 < b - a < d(g)$. Then $g'/g = -1$ at $a, b$. Since $g'/g$ is strictly decreasing between consecutive zeros of $g$, some $x_j \in (a, b)$. For $i = 1, \ldots, k-1$, $x_{i+1} - x_i > b - a$ gives $\tfrac{1}{a - x_i} < \tfrac{1}{b - x_{i+1}}$. Summing:
\[
\frac{g'(a)}{g(a)} = \sum_{i=1}^{k-1} \frac{1}{a - x_i} + \underbrace{\frac{1}{a - x_k}}_{< 0} < \sum \frac{1}{b - x_{i+1}} + \underbrace{\frac{1}{b - x_1}}_{> 0} = \frac{g'(b)}{g(b)},
\]
contradiction.

\textbf{Main.} Induct on $m = \deg f$. $m \le 1$ trivial. For $m \ge 2$: WLOG monic; by induction, $f'$ eventually has $m-1$ distinct real zeros; WLOG $f_n'$ always has $m - 1$ distinct zeros $x_1^{(n)} > \cdots > x_{m-1}^{(n)}$. Apply Rolle to $e^x f_n'$: between consecutive zeros of $f_n'$, $f_{n+1}'$ has exactly one zero. Hence interlacing: $x_j^{(n)} > x_j^{(n+1)} > x_{j+1}^{(n)}$.

\textbf{Lemma 2.} $\lim_n f_n(x_j^{(n)}) = -\infty$ for $j$ odd, $+\infty$ for $j$ even. Once this holds, all local maxima are positive and minima negative, so $f_n$ has $m$ distinct roots.

\emph{Proof sketch.} Let $d = \min(d(f'), 1)$, $\varepsilon = (m-1) d^{m-1}/m^{m-1}$. Write $f = f_n$, $b = x_j^{(n)}$ ($j$ even). Pick $a$ with $d \le b - a \le 1$ and $f'$ nonzero on $(a, b)$; set $\xi$ with $b - \xi = (b - a)/m$. Then $f''(\xi)/f'(\xi) < 0$ (from the rational expression), so $f'' < 0$ on $(\xi, b)$. Therefore
\[
f(b) - f(\xi) \le (b - \xi) f'(\xi),
\]
giving $f(\xi) + f'(\xi) \ge f(b) + (1 - 1/m) f'(\xi)$. Bound $f'(\xi) \ge d^{m-1}/m^{m-2}$, yielding $f(\xi) + f'(\xi) \ge f(b) + \varepsilon$. Combined with $x_j^{(n+1)} > b$: $f_{n+1}(x_j^{(n+1)}) \ge f_n(x_j^{(n)}) + \varepsilon$.
