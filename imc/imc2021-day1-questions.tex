\section*{Problem 1}
Let $A$ be a real $n \times n$ matrix such that $A^3 = 0$.
\begin{itemize}
\item[(a)] Prove that there is a unique real $n \times n$ matrix $X$ satisfying $X + AX + XA^2 = A$.
\item[(b)] Express $X$ in terms of $A$.
\end{itemize}

\section*{Problem 2}
Let $n$ and $k$ be fixed positive integers, and let $a$ be an arbitrary non-negative integer. Choose a random $k$-element subset $X$ of $\{1, 2, \ldots, k+a\}$ uniformly, and independently a random $n$-element subset $Y$ of $\{1, \ldots, k+n+a\}$ uniformly. Prove that $\mathbb{P}(\min(Y) > \max(X))$ does not depend on $a$.

\section*{Problem 3}
A positive real number $d$ is \emph{good} if there exists an infinite sequence $a_1, a_2, \ldots \in (0, d)$ such that for each $n$, the points $a_1, \ldots, a_n$ partition $[0, d]$ into segments of length at most $1/n$ each. Find $\sup\{d \mid d \text{ is good}\}$.

\section*{Problem 4}
Let $f: \mathbb{R} \to \mathbb{R}$ be a function. Suppose that for every $\varepsilon > 0$, there exists $g: \mathbb{R} \to (0, \infty)$ such that for every pair $(x, y)$,
\[ \text{if } |x - y| < \min\{g(x), g(y)\}, \text{ then } |f(x) - f(y)| < \varepsilon. \]
Prove that $f$ is the pointwise limit of a sequence of continuous $\mathbb{R} \to \mathbb{R}$ functions.
