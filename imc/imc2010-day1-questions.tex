\section*{Problem 1}
Let $0 < a < b$. Prove that
\[
\int_a^b (x^2 + 1) e^{-x^2} \, dx \ge e^{-a^2} - e^{-b^2}.
\]

\section*{Problem 2}
Compute the sum of the series
\[
\sum_{k=0}^\infty \frac{1}{(4k+1)(4k+2)(4k+3)(4k+4)} = \frac{1}{1 \cdot 2 \cdot 3 \cdot 4} + \frac{1}{5 \cdot 6 \cdot 7 \cdot 8} + \cdots.
\]

\section*{Problem 3}
Define the sequence $x_1, x_2, \ldots$ by $x_1 = \sqrt{5}$ and $x_{n+1} = x_n^2 - 2$ for $n \ge 1$. Compute
\[
\lim_{n \to \infty} \frac{x_1 \cdot x_2 \cdots x_n}{x_{n+1}}.
\]

\section*{Problem 4}
Let $a, b$ be integers and suppose $n$ is a positive integer for which
\[
\mathbb{Z} \setminus \{ a x^n + b y^n \mid x, y \in \mathbb{Z} \}
\]
is finite. Prove that $n = 1$.

\section*{Problem 5}
Suppose $a, b, c \in [-1, 1]$ satisfy $1 + 2abc \ge a^2 + b^2 + c^2$. Prove that
\[
1 + 2(abc)^n \ge a^{2n} + b^{2n} + c^{2n}
\]
for all positive integers $n$.
