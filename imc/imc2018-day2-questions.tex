\section*{Problem 6}
Let $k$ be a positive integer. Find the smallest $n$ for which there exist nonzero vectors $v_1, \ldots, v_k \in \mathbb{R}^n$ such that $v_i \perp v_j$ for all $|i - j| > 1$.

\section*{Problem 7}
Let $a_0 = 0$ and $a_{n+1}^3 = a_n^2 - 8$. Prove $\sum_{n=0}^\infty |a_{n+1} - a_n|$ converges.

\section*{Problem 8}
Let $\Omega = \{(x,y,z) \in \mathbb{Z}^3 : y + 1 \geq x \geq y \geq z \geq 0\}$. A frog moves along $\Omega$ by unit jumps. For each positive integer $n$, find the number of paths from $(0,0,0)$ to $(n,n,n)$ in exactly $3n$ jumps.

\section*{Problem 9}
Determine all pairs $P(x), Q(x)$ of monic complex polynomials such that $P \mid Q^2 + 1$ and $Q \mid P^2 + 1$.

\section*{Problem 10}
For $R > 1$, let $D_R = \{(a,b) \in \mathbb{Z}^2 : 0 < a^2 + b^2 < R\}$. Compute
\[ \lim_{R \to \infty} \sum_{(a,b) \in D_R} \frac{(-1)^{a+b}}{a^2 + b^2}. \]
