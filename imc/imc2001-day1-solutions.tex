\section*{Problem 1}
Fill an $n\times n$ matrix with entries $1,2,\dots,n^2$ row by row. Choose $n$ entries with one per row and one per column. What are the possible sums?

\subsection*{Solution}
The selection is $\{(j,\sigma(j)) : j=1,\dots,n\}$ for $\sigma\in S_n$, giving sum
\[\sum_{j=1}^n (n(j-1)+\sigma(j)) = n\cdot\tfrac{n(n+1)}{2} - n^2 + \tfrac{n(n+1)}{2} = \tfrac{n(n^2+1)}{2},\]
independent of $\sigma$.

\section*{Problem 2}
Let $r,s,t$ be pairwise coprime positive integers. If $a,b$ are in a commutative group with $a^r=b^s=(ab)^t=e$, prove $a=b=e$. Does this hold in non-commutative groups?

\subsection*{Solution}
Pick $u,v$ with $us+vt=1$. Then $ab=(ab)^{us+vt}=(ab)^{us}=a^{us}(b^s)^u=a^{us}$. Hence $b^r = a^r b^r = (ab)^r = a^{usr} = e$. With $xr+ys=1$, $b=(b^r)^x(b^s)^y = e$. Similarly $a=e$.

Non-commutative counterexample: in $S_7$, $a=(1\,2\,3)$, $b=(3\,4\,5\,6\,7)$, $ab=(1\,2\,3\,4\,5\,6\,7)$, so $a^3=b^5=(ab)^7=e$.

\section*{Problem 3}
Find $\displaystyle\lim_{t\to 1^-}(1-t)\sum_{n=1}^\infty \frac{t^n}{1+t^n}$.

\subsection*{Solution}
\begin{align*}
\lim_{t\to 1^-}(1-t)\sum_{n=1}^\infty \frac{t^n}{1+t^n} &= \lim_{t\to 1^-}\frac{1-t}{-\ln t}\cdot(-\ln t)\sum_{n=1}^\infty\frac{1}{1+e^{-n\ln t}}\\
&= \lim_{h\to 0^+} h\sum_{n=1}^\infty \frac{1}{1+e^{nh}} = \int_0^\infty \frac{dx}{1+e^x} = \ln 2.
\end{align*}

\section*{Problem 4}
Let $p(x)$ be a degree-$n$ polynomial with coefficients in $\{-1,0,1\}$, divisible by $(x-1)^k$. Let $q$ be a prime with $\frac{q}{\ln q}<\frac{k}{\ln(n+1)}$. Prove the non-trivial $q$-th roots of unity are roots of $p$.

\subsection*{Solution}
Let $p(x)=(x-1)^k r(x)$, $\varepsilon_j=e^{2\pi i j/q}$. Since $\Phi_q(x)=\prod(x-\varepsilon_j)$ is irreducible over $\mathbb{Q}$, either all $\varepsilon_j$ are roots of $r$ or none.

If none, $N:=\prod_{j=1}^{q-1} r(\varepsilon_j)\in\mathbb{Z}\setminus\{0\}$, so $|N|\ge 1$. Since $|p(\varepsilon_j)|\le n+1$ and $\prod(1-\varepsilon_j)=q$,
\[(n+1)^{q-1}\ge \prod|p(\varepsilon_j)| = q^k \cdot |N| \ge q^k,\]
so $(q-1)\ln(n+1)\ge k\ln q$, contradicting the hypothesis.

\section*{Problem 5}
Let $A$ be $n\times n$ complex with $A\ne \lambda I$ for any $\lambda$. Prove $A$ is similar to a matrix with at most one nonzero diagonal entry.

\subsection*{Solution}
Induct on $n$. For $n=2$, $A=\begin{pmatrix}a&b\\c&d\end{pmatrix}$. If $b\ne 0$, conjugate by $\begin{pmatrix}1&0\\a/b&1\end{pmatrix}$:
\[\begin{pmatrix}1&0\\a/b&1\end{pmatrix}\begin{pmatrix}a&b\\c&d\end{pmatrix}\begin{pmatrix}1&0\\-a/b&1\end{pmatrix} = \begin{pmatrix}0&b\\c-ad/b&a+d\end{pmatrix}.\]
Symmetric if $c\ne 0$. If $b=c=0$ and $a\ne d$, conjugate by $\begin{pmatrix}1&1\\0&1\end{pmatrix}$ to break symmetry, then proceed.

For $n>2$: write $A=\begin{pmatrix}A'&*\\ *&\beta\end{pmatrix}$ with $A'$ of size $n-1$, $A'\ne\lambda' I$ (WLOG). Induction gives $P$ with $P^{-1}A'P$ having diagonal $(0,\dots,0,\alpha)$. Block-conjugate $A$ by $\mathrm{diag}(P,1)$ to get $B$ with diagonal $(0,\dots,0,\alpha,\beta)$. Now view $B$ as $\begin{pmatrix}0&*\\ *&C\end{pmatrix}$ where $C$ is $(n-1)\times(n-1)$ with diagonal $(0,\dots,0,\alpha,\beta)$. Apply induction to $C$ (assuming $C\ne\lambda I$) to reduce its diagonal to $(0,\dots,0,\gamma)$.

The step can fail only when $n-1=2$ and the reduction forces
\[P^{-1}AP = \begin{pmatrix}0&a&b\\c&d&0\\e&0&d\end{pmatrix},\; d\ne 0.\]
Not all of $a,b,c,e$ are zero. If $b\ne 0$, conjugate by $\begin{pmatrix}1&0&0\\0&1&0\\1&0&1\end{pmatrix}$ yielding
\[\begin{pmatrix}-b&a&b\\ c&d&0\\ e-b-d&a&b+d\end{pmatrix}.\]
Its trace is $d$; applying the 2-case reduction again gives diagonal $(0,d-b,d+b)$, finishing. Cases $a,c,e\ne 0$ are similar.

\section*{Problem 6}
Differentiable $a,b,f,g:\mathbb{R}\to\mathbb{R}$ with $f\ge 0$, $f'\ge 0$, $g>0$, $g'>0$, $a(x)\to A>0$, $b(x)\to B>0$, $f,g\to\infty$, and $\frac{f'}{g'}+a\cdot\frac{f}{g}=b$. Prove $\lim f/g = B/(A+1)$.

\subsection*{Solution}
Fix $0<\varepsilon<A$. For $x$ large, $|a-A|<\varepsilon$, $|b-B|<\varepsilon$, so
\[B-\varepsilon < \frac{f'}{g'}+(A+\varepsilon)\frac{f}{g}.\]
Using $(fg^A)' = f'g^A + Afg^{A-1}g'$ and $(g^{A+1})' = (A+1)g^A g'$, some algebra gives
\[\frac{(fg^A)'}{(g^{A+1})'} > \frac{A(B-\varepsilon)}{(A+\varepsilon)(A+1)},\]
and similarly $< \frac{A(B+\varepsilon)}{(A-\varepsilon)(A+1)}$ for an upper bound. Letting $\varepsilon\to 0$,
\[\lim \frac{(fg^A)'}{(g^{A+1})'} = \frac{B}{A+1}.\]
By l'Hôpital, $\lim \frac{fg^A}{g^{A+1}} = \lim \frac{f}{g} = \frac{B}{A+1}$.
